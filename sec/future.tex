\section{Future work}
\label{ch:future_work}

\structure{Data requirements}
\outline{Collect more data, several hundreds of thousands of examples per class}
\outline{Allows testing techniques like undersampling}
\outline{Also, deeper architectures could be evaluated}
\outline{Multi-lingual models are possibility as GloVe provide word vectors for
different languages}

\structure{Model architectures}
\outline{Hyperparameter tuning and model architectures were tested manually}
\outline{Evolve NN architectures using automated approaches, e.g., AutoML}

\structure{Interpretability}
\outline{Try to interpret model features}
\outline{Special focus should be on features learned from text}
